Ejemplo: Sentencia IF - ELSE 
	
	mi_proceso: process (sel, A, B, C, D) <- Lista de sensibilidad
	begin
		if sel = "00" then
			out <= A;
		elsif sel = "10" then
			out <= B;
		elsif sel = "01" then
			out <= C;
		else 
			out <= D;
		end if;
	end process mi_proceso;

Ejemplo: Sentencia CASE
	proceso: process (sel, A, B, C, D)
	begin
		case (sel) is
			when "00" => out <= A;
			when "01" => out <= B;
			when "10" => out <= C;
			when others => out <= D;
		end case;
	end process proceso;


-- EJEMPLO COMBINADO DE SENTENCIA IF - CASE

proceso: process (Reset, clock)
begin 
	if reset = '1' then
		out <= '0';
	elsif rising_edge (clock) then
		case (sel) is
			when "00" => out <= A;
			when "01" => out <= B;
			when "10" => out <= C;
			when others => out <= D;
		end case;
	end if;
end process proceso;
